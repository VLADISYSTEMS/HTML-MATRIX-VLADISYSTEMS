/* -----------------------------
JavaScript para mostrar un resumen de la encuesta al enviar
----------------------------- */

// Seleccionamos el formulario y agregamos un evento al enviarlo
document.getElementById('encuesta-form').addEventListener('submit', function(e){

    e.preventDefault(); // Evita recargar la p√°gina

    // Capturamos los valores de los campos
    const nombre = document.getElementById('nombre').value;
    const email = document.getElementById('email').value;
    const satisfaccion = document.querySelector('input[name="satisfaccion"]:checked')?.value;
    const aspectos = Array.from(document.querySelectorAll('input[name="aspectos"]:checked'))
                          .map(cb => cb.value)
                          .join(', ');
    const comentarios = document.getElementById('comentarios').value;

    // Mostramos el resultado en pantalla
    document.getElementById('resultado').innerHTML = `
        <h3>Resumen de la Encuesta</h3>
        <p><strong>Nombre:</strong> ${nombre}</p>
        <p><strong>Correo:</strong> ${email}</p>
        <p><strong>Satisfacci√≥n:</strong> ${satisfaccion || 'Sin seleccionar'}</p>
        <p><strong>Aspectos favoritos:</strong> ${aspectos || 'Ninguno seleccionado'}</p>
        <p><strong>Comentarios:</strong> ${comentarios || 'Sin comentarios'}</p>
    `;
});
</script>
